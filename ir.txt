const int numCommands = sizeof(commands) / sizeof(commands[0]);
int currentCommand = 0;
bool spamming = false;          // Флаг спама
unsigned long lastSendTime = 0;
const unsigned long sendInterval = 500; // Интервал между отправками (мс)

IRsend irsend(IR_LED_PIN);

void setup() {
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
  Serial.println("Universal TV Power Off Remote - SPAM MODE");
  Serial.println("Press button to START/STOP spamming");
  Serial.print("Current TV: ");
  Serial.println(commands[currentCommand].name);
}

void loop() {
  // Проверка нажатия кнопки (переключение режима спама)
  static bool lastButtonState = HIGH;
  bool currentButtonState = digitalRead(BUTTON_PIN);
  
  // Обработка нажатия (фронт сигнала)
  if (lastButtonState == HIGH && currentButtonState == LOW) {
    delay(50); // Антидребезг
    if (digitalRead(BUTTON_PIN) == LOW) {
      spamming = !spamming; // Переключение спама
      
      if (spamming) {
        Serial.println(">>> SPAM STARTED <<<");
        digitalWrite(LED_PIN, HIGH); // Светодиод горит во время спама
      } else {
        Serial.println(">>> SPAM STOPPED <<<");
        digitalWrite(LED_PIN, LOW);
      }
      
      // Ждем отпускания кнопки
      while(digitalRead(BUTTON_PIN) == LOW) {
        delay(10);
      }
    }
  }
  lastButtonState = currentButtonState;
  
  // Спам-режим
  if (spamming) {
    unsigned long now = millis();
    if (now - lastSendTime >= sendInterval) {
      sendCurrentSignal();
      lastSendTime = now;
    }
  }
}

void sendCurrentSignal() {
  Serial.print(".");
  
  if (commands[currentCommand].isRaw) {
    if (commands[currentCommand].rawData != nullptr) {
      irsend.sendRaw(commands[currentCommand].rawData, 
                     commands[currentCommand].rawLen, 
                     38);
    }
  } else {
    switch(commands[currentCommand].protocol) {
      case SAMSUNG:
        irsend.sendSAMSUNG(commands[currentCommand].command, 32);
        break;
      case RC5:
        irsend.sendRC5(commands[currentCommand].command, 12);
        break;
      case SIRC:
        irsend.sendSony(commands[currentCommand].command, 12);
        break;
      case NEC:
        irsend.sendNEC(commands[currentCommand].command, 32);
        break;
      case RC6:
        irsend.sendRC6(commands[currentCommand].command, 20);
        break;
      default:
        return;
    }
  }
}

void printCurrentTV() {
  Serial.println(commands[currentCommand].name);
}
